134  double BinaryMutation(double inputValue) {
135  double probability = 2.0; //default probability of mutation is 2%
136  srand (time(NULL));
137  std::random_device generator_device;
138  std::mt19937_64 gen(generator_device());
139  std::uniform_real_distribution<double> distribution_device(0.0,100.0);
140  double number_extracted = distribution_device(gen);
141 
142  double RangeMAX = 10.0;
143  double integer_weight = 0;
144 
145 
146  //---FASE 1: conversione nel range -128/127
147  if (inputValue >0) integer_weight = ((RangeMAX-inputValue) *256) /(RangeMAX*2);
148  else integer_weight = (((RangeMAX-inputValue) *256) /(RangeMAX*2))-1;
149 
150 
151  //FASE 2: Ricavo il byte corrispettivo
152  int b[8];
153  int val = (int)integer_weight;
154 
155  for(int Contatore = 0; Contatore<8; Contatore++){
156  b[Contatore]= val % 2;
157  val = val /2;
158  }
159 
160  //FASE 3: muto il byte agendo sui singoli bit
161  for(int ContaBit =0; ContaBit<8;ContaBit++){
162  number_extracted = distribution_device(gen);
163  if (number_extracted <= probability){
164  if (b[ContaBit] ==0) b[ContaBit]=1;
165  else b[ContaBit]=0;
166  }
167  }
168 
169  //FASE 4: ritrasformo il Byte nel Range 255
170  integer_weight =0;
171  integer_weight+=b[0]*1;
172  integer_weight+=b[1]*2;
173  integer_weight+=b[2]*4;
174  integer_weight+=b[3]*8;
175  integer_weight+=b[4]*16;
176  integer_weight+=b[5]*32;
177  integer_weight+=b[6]*64;
178  integer_weight+=b[7]*128;
179 
180  //FASE 5: trasformo il numero dal range 255 al Double
181  if (integer_weight<0) integer_weight+=256;
182 
183  inputValue= (integer_weight + 1)/256;
184  inputValue=RangeMAX -(inputValue*(RangeMAX*2));
185  return inputValue;
186  }
